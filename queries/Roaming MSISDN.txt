SELECT 
	DISTINCT 
	MSISDN,
	THEIR_PRIMARY_PMN_TADIG_CODE,
	C.COUNTRY
from 
	ATNI_PROD.GTT_REPORTING.TAP_INCOLLECT A
  	 LEFT JOIN ATNI_DEV.GTT_REPORTING.COUNTRY_CODES C 
 ON left(A.THEIR_PRIMARY_PMN_TADIG_CODE,3) = trim(REPLACE(C.alpha3_code,'"', ''))
WHERE 
	MY_PMN_TADIG_CODE='GUYGT'
	--AND THEIR_PRIMARY_PMN_TADIG_CODE LIKE 'USA%'
	AND TRY_TO_DATE(CALL_EVENT_START_TIMESTAMP) >= to_date('2020-01-01','YYYY-MM-DD')
