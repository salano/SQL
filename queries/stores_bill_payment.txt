SELECT
	to_char(PAYMENT_TIMESTAMP, 'YYYY-MM' ) MTH,
	PAYMENT_LOCATION_NAME,
	PAYMENT_CHANNEL_CATGORY,
	PAYMENT_METHOD_CODE,
	count(NUMBER_OF_PAYMENTS) NUMBER_OF_PAYMENTS,
	sum(PAYMENT_AMOUNT_USD) PAYMENT_AMOUNT_USD,
	sum(PAYMENT_AMOUNT_GYD)PAYMENT_AMOUNT_GYD 
FROM 
	ATNI_PROD.GTT_REPORTING.PAYMENT 
WHERE 
	PAYMENT_TYPE_NAME ='Bill Payment' 
	AND PAYMENT_LOCATION_NAME IN ('ANNA REGINA','YURABALLI ST','REGENT AND CAMP STREETS RETAIL STORE','New Amsterdam','FRANCHISE - NEW AMSTERDAM','6 STRAND',
	'BETERVERWAGTING','PARIKA RETAIL STORE')
	and date(PAYMENT_TIMESTAMP) >= date_trunc('MONTH', ADD_MONTHS(current_date, -3)) 
GROUP BY 1,2,3,4
ORDER BY 1,2
