SELECT 
DISTINCT 
   a.CUSTOMER_NAME,
   a.CCT_ID AS PHONE_NUMBER,
   A.EXCHANGE_NAME,
   B.SERIAL AS MODEM_MAC
FROM
 ATNI_PROD.GTT_REPORTING.GTT_DSL_DATA_RATE A
 JOIN ATNI_DEV.GTT_REPORTING.TMP_DSL_MODEMS B
 ON A.NODE_IP = B.NODE  AND A.SLOT = B.SLOT AND A.PORT = B.PORT 
WHERE date(CREATED_DATE) = date(current_date)
