SELECT 
	DATESTART AS disconnect_date
	,sv.BILL_FNAME 
	,sv.BILL_LNAME
	,sv.BILL_COMPANY 
	,ACCOUNTNUMBER 
	,BID AS Phone_number
	,CUST_EMAIL Email
    ,c.statement_to_email
  	,CUST_PHONE1
  	,CUST_PHONE2
  	,CONTACT1_PHONE
  	,CONTACT2_PHONE
	,CUSTOMER_TYPE_NAME 
	, LINE_OF_SERVICE 
	, SUBLINE_OF_SERVICE
	,SUBLINE_OF_SERVICE_SEGMENT 
	,PRODUCT_NAME
	,TECHNOLOGY_NAME 
FROM 
	ATNI_DEV.GTT_REPORTING.SV_GTT_BUSINESS_SUBSCRIBERS sv
	JOIN ATNI_PROD.GTT_REPORTING.CMF_EXTRACT c 
		on sv.ACCOUNTNUMBER = c.ACCOUNT_NO AND date(c.EXTRACT_FILE_DATE) = date(CURRENT_DATE())
WHERE
	REPORT_MONTH= DATE_TRUNC('MONTH',CURRENT_DATE())
	AND NET_DEACTIVATIONS <> 0;
