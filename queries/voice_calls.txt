SELECT 

	C.EXTERNAL_ID MSISDN ,
	CASE WHEN EXTERNAL_ID_TYPE = 1 THEN 'Mobile'
		WHEN EXTERNAL_ID_TYPE IN (9,27) THEN 'Landline'
		WHEN EXTERNAL_ID_TYPE = 1 THEN 'Fibre'
	END Product,
	c.TRANS_DT,
	C.PRIMARY_UNITS duration_sec
FROM 

	ATNI_PROD.GTT_REPORTING.CDR_DATA C
WHERE 
	 date(c.TRANS_DT)  >= dateadd(day, -61, current_date)
	and c.APPLICATION_ID  =1 and EXTERNAL_ID_TYPE IN (1,9,27,37)
