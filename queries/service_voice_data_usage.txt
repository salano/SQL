SELECT 
date_trunc('DAY',DATESTART) DATESTART,
   EXTERNAL_ID,
   ACCOUNT_NO,
   SUBSCR_NO,
   CASE WHEN A.ORIGINATINGTECHNOLOGYCODE = 8 THEN 'Wireline'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 12 THEN 'DSL'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 4 THEN 'Mobile'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 15 THEN 'Leased Line'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 17 THEN 'Calling Card'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 18 THEN 'WiMax'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 22 THEN 'Interconnect'
            WHEN A.ORIGINATINGTECHNOLOGYCODE = 24 THEN 'GPON'
           END SERVICENAME,
   sum(ratedduration) voice_duration,
   sum(ratedKB) ratedKB
FROM 
	ATNI_PROD.GTT_REPORTING.SV_GTT_RAA_USAGE_DETAIL A
WHERE 
	date(A.DATESTART) >= to_date('2021-01-01','YYYY-MM-DD')
GROUP BY 1,2,3,4,5
UNION 
SELECT 
  date_trunc('DAY',date(A.DATE_PERIOD)) DATESTART,
  A.MSIDN AS EXTERNAL_ID,
  E.ACCOUNT_NO,
  E.SUBSCR_NO,
  'GPON Wireline' SERVICENAME,
  sum(C.PRIMARY_UNITS) voice_duration,
  sum(A.TOTAL_TRANSMITTED_BYTES/1024) ratedKB
 FROM
	CDR.GTT_SANDVINE_WIRELINE A 
	LEFT JOIN ATNI_PROD.GTT_REPORTING.EXTERNAL_ID_EQUIP_MAP_EXTRACT E on try_TO_NUMERIC(CONCAT('592',A.MSIDN)) = try_TO_NUMERIC(E.EXTERNAL_ID) AND date(E.EXTRACT_FILE_DATE) = date(CURRENT_DATE()) AND E.IS_CURRENT = 1
	LEFT JOIN ATNI_PROD.GTT_REPORTING.CDR_DATA C  ON E.EXTERNAL_ID = C.EXTERNAL_ID and C.EXTERNAL_ID_TYPE IN (27,37) and c.APPLICATION_ID  =1
WHERE 
	date(A.DATE_PERIOD) >= to_date('2021-01-01','YYYY-MM-DD')
	AND C.TRANS_DT >= to_date('2021-01-01','YYYY-MM-DD')
GROUP BY 1,2,3,4,5
