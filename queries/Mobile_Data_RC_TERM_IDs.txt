--Mobile Data RC TERM IDs
SELECT 
	RTV.RC_TERM_ID,
	RTV.DISPLAY_VALUE,
	RTR.DATE_ACTIVE,
	RTR.DATE_INACTIVE 
FROM RC_TERM_VALUES RTV
	JOIN RC_TERM_REF RTR
	ON RTR.RC_TERM_ID = RTV.RC_TERM_ID
	AND RTR.RESELLER_VERSION_ID = (SELECT MAX(RESELLER_VERSION_ID) FROM RC_TERM_REF)
	AND RTR.EXTERNAL_ID_TYPE = 1
--	AND (RTR.DATE_INACTIVE IS NULL OR DATE(RTR.DATE_INACTIVE) > CURRENT_DATE)
WHERE RTV.RESELLER_VERSION_ID = (SELECT MAX(RESELLER_VERSION_ID) FROM RC_TERM_VALUES)
AND UPPER(RTV.DISPLAY_VALUE) LIKE '%DATA%'
AND UPPER(RTV.DISPLAY_VALUE) NOT LIKE '%DATA PORT%'
ORDER BY 1
--LIMIT 100
;
